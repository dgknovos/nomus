109      /**
110       * Parse the current document.  (This is the principal 
111       * method called by a CorpusController.)
112       */
113      public void execute() throws ExecutionException {
114        annotationSet  = convertASName(annotationSetName);
115        
116        if (debugMode) {
117          System.out.println("Parsing document: " + document.getName());
118        }
119    
120        if (useMapping && (! mappingLoaded) ) {
121          System.err.println("Warning: no mapping loaded!");
122        }
123        
124        
125        if (addConstituentAnnotations || addDependencyFeatures || addDependencyAnnotations) {
126          parseSentences();
127        }
128        else {
129          System.err.println("There is nothing for the parser to do.");
130          System.err.println("Please enable at least one of the \"add...\" options.");
131        }
132      }

135      /**
136       * Initialize the Parser resource.  In particular, load the trained data
137       * file.
138       */
139      public Resource init() throws ResourceInstantiationException {
140        instantiateStanfordParser();
141        if (mappingFile != null) {
142          loadTagMapping(mappingFile);
143        }
144        offsetComparator = new OffsetComparator();
145        super.init();
146        
147        if(tlppClass == null || tlppClass.equals("")) {
148          throw new ResourceInstantiationException(
149                  "TLPP class name must be specified");
150        }
151        try {
152          Class<?> tlppClassObj =
153                  Class.forName(tlppClass, true, Gate.getClassLoader());
154          if(!TreebankLangParserParams.class.isAssignableFrom(tlppClassObj)) {
155            throw new ResourceInstantiationException(tlppClassObj
156                    + " does not implement "
157                    + TreebankLangParserParams.class.getName());
158          }
159          TreebankLangParserParams tlpp =
160                  TreebankLangParserParams.class.cast(tlppClassObj.newInstance());
161          gsf = tlpp.treebankLanguagePack().grammaticalStructureFactory();
162        }
163        catch(ClassNotFoundException e) {
164          throw new ResourceInstantiationException("Class " + tlppClass
165                  + " not found", e);
166        }
167        catch(InstantiationException e) {
168          throw new ResourceInstantiationException("Error creating TLPP object", e);
169        }
170        catch(IllegalAccessException e) {
171          throw new ResourceInstantiationException("Error creating TLPP object", e);
172        }
173        return this;
174      }

187      /**
188       * Find all the Sentence annotations and iterate through them, parsing one
189       * sentence at a time and storing the result in the output AS. (Sentences are
190       * scanned for Tokens. You have to run the ANNIE tokenizer and splitter before
191       * this PR.)
192       */
193      @SuppressWarnings("unchecked")
194      private void parseSentences() { 
195        List<Annotation> sentences = new ArrayList<Annotation>(annotationSet.get(inputSentenceType));
196        java.util.Collections.sort(sentences, offsetComparator);
197        Iterator<Annotation> sentenceIter = sentences.iterator();
198    
199        Tree tree;
200        int debugNbrS, debugS;
201        debugS = 0;
202        debugNbrS = sentences.size();
203    
204        while (sentenceIter.hasNext()) {
205          debugS++;
206          tree = parseOneSentence(sentenceIter.next(), debugS);
207          
208          // Here null is the result from an empty Sentence.
209          if (tree != null) {
210            if (addConstituentAnnotations || addPosTags) {
211              annotatePhraseStructureRecursively(tree, tree);
212            }
213    
214            if (addDependencyFeatures || addDependencyAnnotations) {
215              annotateDependencies(tree);
216            }
217    
218            if (debugMode) {
219              System.out.println("Parsed sentence " + debugS + " of " + debugNbrS);
220            }
221          }
222          
223          else if (debugMode) {
224            System.out.println("Ignored empty sentence " + debugS + " of " + debugNbrS);
225          }
226          
227        }
228      }

231      /**
232       * Generate the special data structure for one sentence and pass the List of
233       * Word to the parser.
234       * 
235       * @param sentence
236       *          the Sentence annotation
237       * @param s
238       *          sentence number of debugging output
239       * @param ofS
240       *          total number of sentences for debugging output
241       * @return  null if the sentence is empty
242       */
243      private Tree parseOneSentence(Annotation sentence, int sentenceNo) {
244        Tree result = null;
245        
246        stanfordSentence = new StanfordSentence(sentence, inputTokenType, annotationSet, reusePosTags);
247    
248        /* Ignore an empty Sentence (sometimes the regex splitter can create one
249         * with no Token annotations in it).
250         */
251        if ( stanfordSentence.isNotEmpty() ) {
252          List<Word> wordList = stanfordSentence.getWordList();
253    
254          if (reusePosTags) {
255            int nbrMissingTags = stanfordSentence.numberOfMissingPosTags();
256            if (nbrMissingTags > 0)  {
257              double percentMissing = Math.ceil(100.0 * ((float) nbrMissingTags) /
258                      ((float) stanfordSentence.numberOfTokens()) );
259              System.err.println("Warning (sentence " + sentenceNo + "): " + (int) percentMissing 
260                      + "% of the Tokens are missing POS tags." );
261            }
262          }
263    
264          stanfordParser.parse(wordList);
265          result = stanfordParser.getBestParse();
266        }
267        
268        return result;
269      }

272      /**
273       * Generate a SyntaxTreeNode Annotation corresponding to this Tree.  Work 
274       * recursively so that the annotations are actually generated from the 
275       * bottom up, in order to build the consists list of annotation IDs.
276       * 
277       * @param tree  the current subtree
278       * @param rootTree  the whole sentence, used to find the span of the current subtree
279       * @return a GATE Annotation of type "SyntaxTreeNode"
280       */
281      protected Annotation annotatePhraseStructureRecursively(Tree tree, Tree rootTree) {
282        Annotation annotation = null;
283        Annotation child;
284        String label   = tree.value();
285    
286        List<Tree> children = tree.getChildrenAsList();
287    
288        if (children.size() == 0) {
289          return null;
290        }
291        /* implied else */
292    
293        /* following line generates ClassCastException
294         *          IntPair span = tree.getSpan();
295         * edu.stanford.nlp.ling.CategoryWordTag
296         * at edu.stanford.nlp.trees.Tree.getSpan(Tree.java:393)
297         * but I think it's a bug in the parser, so I'm hacking 
298         * around it as follows. */
299        int startPos = Trees.leftEdge(tree, rootTree);
300        int endPos   = Trees.rightEdge(tree, rootTree);
301        
302        Long startNode = stanfordSentence.startPos2offset(startPos);
303        Long endNode   = stanfordSentence.endPos2offset(endPos);
304    
305        List<Integer> consists = new ArrayList<Integer>();
306    
307        Iterator<Tree> childIter = children.iterator();
308        while (childIter.hasNext()) {
309          child = annotatePhraseStructureRecursively(childIter.next(), rootTree);
310          if  ( (child != null)  &&
311            (! child.getType().equals(inputTokenType) )) {
312            consists.add(child.getId());
313          }
314        }
315        annotation = annotatePhraseStructureConstituent(startNode, endNode, label, consists, tree.depth());
316    
317        return annotation;
318      }

322      /**
323       * Record one constituent as an annotation.
324       * 
325       * @param startOffset
326       * @param endOffset
327       * @param label
328       * @param consists
329       * @param depth
330       * @return
331       */
332      private Annotation annotatePhraseStructureConstituent(Long startOffset, Long endOffset, String label, 
333        List<Integer> consists, int depth) {
334        Annotation phrAnnotation = null;
335        Integer phrID;
336    
337        try {
338          String cat;
339          if (useMapping && mappingLoaded) {
340            cat  = translateTag(label);
341          }
342          else {
343            cat = label; 
344          }
345          
346          if (addConstituentAnnotations) {
347            String text = document.getContent().getContent(startOffset, endOffset).toString();
348            FeatureMap fm = gate.Factory.newFeatureMap();
349            fm.put(PSG_TAG_FEATURE, cat);
350            fm.put("text", text);
351    
352            /* Ignore empty list features on the token-equivalent annotations. */
353            if (consists.size() > 0) {
354              fm.put("consists", consists);
355            }
356    
357            phrID  = annotationSet.add(startOffset, endOffset, OUTPUT_PHRASE_TYPE, fm);
358            phrAnnotation = annotationSet.get(phrID);
359            recordID(annotationSet, phrID);
360          }
361    
362          if ( addPosTags && (depth == 1) ) {
363            /* Expected to be a singleton set! */
364            AnnotationSet tokenSet = annotationSet.get(inputTokenType, startOffset, endOffset);
365            if (tokenSet.size() == 1) {
366              Annotation token = tokenSet.iterator().next();
367    
368              /* Add POS tag to token.  
369               * (Note: GATE/Hepple uses "(" and ")" for Penn/Stanford's
370               * "-LRB-" and "-RRB-". */
371              String hepCat = StanfordSentence.unescapePosTag(cat);
372              token.getFeatures().put(POS_TAG_FEATURE, hepCat);
373              
374            }
375            else {
376              System.err.println("Found a tokenSet with " + tokenSet.size() + " members!");
377            }
378          }
379        }
380        catch (InvalidOffsetException e) {
381          e.printStackTrace();
382        }
383        
384        return phrAnnotation;
385      }

389      @SuppressWarnings("unchecked")
390      private void annotateDependencies(Tree tree) {
391        GrammaticalStructure gs = gsf.newGrammaticalStructure(tree);
392        Collection<TypedDependency> deps = gs.typedDependencies();
393        String dependencyKind;
394        FeatureMap depFeatures;
395        Integer dependentTokenID, governorTokenID;
396        List<Integer> argList;
397        Long offsetLH0, offsetRH0, offsetLH1, offsetRH1, depLH, depRH;
398        Annotation governorToken, dependentToken;
399    
400        for(TypedDependency d : deps) {
401          if(debugMode) {
402            System.out.println(d);
403          }
404    
405          int governorIndex = ((HasIndex) d.gov().label()).index() - 1;
406          governorToken  = stanfordSentence.startPos2token(governorIndex);
407          
408          int dependentIndex = ((HasIndex) d.dep().label()).index() - 1;
409          dependentToken = stanfordSentence.startPos2token(dependentIndex);
410    
411          dependencyKind = d.reln().toString();
412          governorTokenID = governorToken.getId();
413          dependentTokenID = dependentToken.getId();
414    
415          if (addDependencyFeatures) {
416            List<DependencyRelation> depsForTok =
417              (List<DependencyRelation>) governorToken.getFeatures().get(dependenciesFeature);
418    
419            if(depsForTok == null) {
420              depsForTok = new ArrayList<DependencyRelation>();
421              governorToken.getFeatures().put(dependenciesFeature, depsForTok);
422            }
423    
424            depsForTok.add(new DependencyRelation(dependencyKind, dependentTokenID));
425          }
426          
427          
428          if (addDependencyAnnotations) {
429            depFeatures = gate.Factory.newFeatureMap();
430            argList = new ArrayList<Integer>();
431            argList.add(governorTokenID);
432            argList.add(dependentTokenID);
433            depFeatures.put(DEP_ARG_FEATURE, argList);
434            depFeatures.put(DEP_LABEL_FEATURE, dependencyKind);
435          
436            offsetLH0 = governorToken.getStartNode().getOffset();
437            offsetRH0 = governorToken.getEndNode().getOffset();
438            offsetLH1 = dependentToken.getStartNode().getOffset();
439            offsetRH1 = dependentToken.getEndNode().getOffset();
440          
441            depLH = Math.min(offsetLH0, offsetLH1);
442            depRH = Math.max(offsetRH0, offsetRH1);
443          
444            try {
445              annotationSet.add(depLH, depRH, DEP_ANNOTATION_TYPE, depFeatures);
446            }
447            catch(InvalidOffsetException e) {
448              e.printStackTrace();
449            }
450          }
451        }
452      }

468      private void loadTagMapping(File mappingFile)  { 
469        tagMap = new HashMap<String, String>();
470        mappingLoaded = false;
471    
472        try {
473          if (mappingFile.exists() && mappingFile.canRead()) {
474    
475            BufferedReader br = new BufferedReader(new FileReader(mappingFile));
476            String line = "";
477    
478            // read until it reaches to an end of the file
479            while((line = br.readLine()) != null) {
480              // two columns delimited by whitespace 
481              String [] data = line.split("\\s+", 2);
482    
483              // are there key and value available
484              if(data == null || data.length < 2) {
485                continue;
486              } else {
487                // and add it to the map
488                tagMap.put(data[0].trim(), data[1].trim());
489              }
490            }
491    
492            br.close();
493          }
494    
495          else {
496            System.err.println("Can't find or read mapping file " 
497              + mappingFile.getPath() + " so no mappings will be used.");
498          }
499        } 
500        catch(Exception e) {
501          System.err.println("Exception trying to load mapping file "
502            + mappingFile.getPath());
503          e.printStackTrace();
504        }
505    
506        int nbrMapped = tagMap.size();
507        System.out.println("Loaded " + nbrMapped + " mappings from file " + mappingFile);
508        mappingLoaded = (nbrMapped > 0);
509      }

513       * This method stores the annotation ID as a value of feature "ID" on the
514       * relevant annotation. (Mainly to make the ID visible in the GUI for
515       * debugging.)
516       * 
517       * @param annSet
518       * @param annotationID
519       */
520      private void recordID(AnnotationSet annSet, Integer annotationID) {
521        annSet.get(annotationID).getFeatures().put("ID", annotationID);
522      }

525      /**
526       * Translate the tag in the map, or leave it the same if there is no
527       * translation.
528       * 
529       * @param stanfordTag
530       * @return
531       */
532      private String translateTag(String stanfordTag) {
533        String translatedTag = stanfordTag;
534    
535        if (tagMap.containsKey(stanfordTag)) {
536          translatedTag = tagMap.get(stanfordTag);
537        }
538        
539        return translatedTag;
540      }

544        if ((name == null) || name.equals("") ) {
545          return document.getAnnotations();
546        }
547    
548        /* implied else */
549        return document.getAnnotations(name);
550      }
